packageList["xAct"] = {
    "xAct`xPerm`",
    "xAct`xTensor`",
    "xAct`xPert`",
    "xAct`Invar`",
    "xAct`xCoba`",
    "xAct`SymManipulator`",
    "xAct`xTras`",
    "xAct`TexAct`"
};
(*hide the version messages using Block when Needs*)
Block[ {Print},
    Needs/@packageList["xAct"];
];

(*hide the dollar symbols of local variables*)
System`$PrePrint = xAct`xTensor`ScreenDollarIndices;
(*hide the definition messages of objects*)
xAct`xTensor`$DefInfoQ = False;
(*print the covariant derivatives as prefix*)
xAct`xTensor`$CovDFormat = "Prefix";

measure::usage = "Sqrt[-g]";
tensorSim::usage = "x//ContractMetric//Simplification";

measure[metric_?xAct`xTensor`MetricQ] :=
    Sqrt[xAct`xTensor`SignDetOfMetric@metric ToExpression["Det"<>ToString@metric<>"[]"]];

tensorSim[expr_] :=
    expr//xAct`xTensor`ContractMetric//xAct`xTensor`Simplification;
